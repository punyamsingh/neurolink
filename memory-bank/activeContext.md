```typescript
  } catch (error: any) {
    // execSync throws on non-zero exit codes, but we still get the output
    const stdout = error.stdout || "";
    const stderr = error.stderr || "";
    const exitCode = error.status || 1;
    return { stdout, stderr, exitCode };
  }
}
```
